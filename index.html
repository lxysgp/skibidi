const canvas = document.getElementById('game'); const ctx = canvas.getContext('2d'); const hud = document.getElementById('hud'); const log = document.getElementById('log'); const tileSize = 40; const rows = 15; const cols = 15; let level = 1; let swordReady = false; let enemies = []; let items = []; let sword = null; let portal = null; const player = { x: 0, y: 0, hp: 20, maxHp: 20, attack: 3, aura: 0, inventory: [] }; function logMsg(msg) { log.innerHTML += `> ${msg}
`; log.scrollTop = log.scrollHeight; } function updateHUD() { hud.innerHTML = ` Level: ${level}
HP: ${player.hp}/${player.maxHp}
Attack: ${player.attack}
Aura: ${player.aura}
Inventory: ${player.inventory.join(', ') || 'Empty'}
Sword: ${swordReady ? 'Ready! (Space)' : 'Not ready'} `; } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.font = '20px monospace'; for (let y = 0; y < rows; y++) { for (let x = 0; x < cols; x++) { ctx.fillStyle = '#333'; ctx.fillText('.', x * tileSize + 15, y * tileSize + 25); } } for (const item of items) { ctx.fillStyle = 'gold'; ctx.fillText(item.char, item.x * tileSize + 15, item.y * tileSize + 25); } for (const e of enemies) { ctx.fillStyle = e.char === 'B' ? 'magenta' : 'red'; ctx.fillText(e.char, e.x * tileSize + 15, e.y * tileSize + 25); } if (portal) { ctx.fillStyle = 'cyan'; ctx.fillText('P', portal.x * tileSize + 15, portal.y * tileSize + 25); } if (sword) { ctx.fillStyle = 'white'; ctx.fillText('🗡️', sword.x * tileSize + 15, sword.y * tileSize + 25); } ctx.fillStyle = 'lime'; ctx.fillText('@', player.x * tileSize + 15, player.y * tileSize + 25); } function spawnLevel() { enemies = []; items = []; sword = null; portal = null; const enemyChars = ['T', 'M', 'C']; const itemTypes = [ { char: '*', effect: 'aura', value: 1, name: 'Aura Crystal' }, { char: '!', effect: 'attack', value: 2, name: 'Toilet Plunger' }, { char: '%', effect: 'heal', value: 5, name: 'Sigma Shades' }, { char: '$', effect: 'aura', value: 2, name: 'Toilet Gold' } ]; for (let i = 0; i < 6 + level; i++) { enemies.push({ x: rand(cols), y: rand(rows), char: enemyChars[rand(enemyChars.length)], hp: 5 + level * 2 + rand(3), attack: 2 + level }); } enemies.push({ x: cols - 2, y: rows - 2, char: 'B', hp: 12 + level * 3, attack: 4 + level }); for (let i = 0; i < 5; i++) { let it = itemTypes[rand(itemTypes.length)]; items.push({ x: rand(cols), y: rand(rows), ...it }); } player.x = 0; player.y = 0; swordReady = false; sword = null; } function rand(n) { return Math.floor(Math.random() * n); } function pickupItem(x, y) { const item = items.find(i => i.x === x && i.y === y); if (!item) return; items.splice(items.indexOf(item), 1); player.inventory.push(item.name); switch (item.effect) { case 'aura': player.aura += item.value; break; case 'attack': player.attack += item.value; break; case 'heal': player.hp = Math.min(player.maxHp, player.hp + item.value); break; } logMsg(`Picked up ${item.name}!`); } function attackEnemyAt(x, y) { const e = enemies.find(e => e.x === x && e.y === y); if (!e) return; if (!swordReady) { logMsg("Swing your sword first! (Press SPACE)"); return; } e.hp -= player.attack; logMsg(`You hit ${e.char} for ${player.attack} damage!`); if (e.hp <= 0) { logMsg(`${e.char} defeated!`); enemies.splice(enemies.indexOf(e), 1); if (e.char === 'B') { logMsg("You defeated the boss! A portal appears..."); portal = { x: e.x, y: e.y }; } } else { player.hp -= e.attack; logMsg(`${e.char} counterattacks for ${e.attack} damage!`); if (player.hp <= 0) { logMsg("You died... Refresh to try again."); document.removeEventListener('keydown', handleKey); } } sword = null; swordReady = false; updateHUD(); } function movePlayer(dx, dy) { const nx = player.x + dx; const ny = player.y + dy; if (nx < 0 || ny < 0 || nx >= cols || ny >= rows) return; if (portal && nx === portal.x && ny === portal.y) { level++; player.maxHp += 5; player.hp = player.maxHp; logMsg(`Entering Level ${level}...`); spawnLevel(); draw(); updateHUD(); return; } if (enemies.some(e => e.x === nx && e.y === ny)) { attackEnemyAt(nx, ny); return; } player.x = nx; player.y = ny; pickupItem(nx, ny); sword = null; draw(); updateHUD(); } function handleKey(e) { if (e.key === ' ') { swordReady = true; const dirs = [[0,-1],[1,0],[0,1],[-1,0]]; for (const [dx, dy] of dirs) { const sx = player.x + dx; const sy = player.y + dy; if (sx >= 0 && sy >= 0 && sx < cols && sy < rows) { sword = { x: sx, y: sy }; break; } } logMsg("Sword swung! 🗡️"); draw(); updateHUD(); return; } let dx = 0, dy = 0; if (e.key === 'ArrowUp' || e.key === 'w') dy = -1; if (e.key === 'ArrowDown' || e.key === 's') dy = 1; if (e.key === 'ArrowLeft' || e.key === 'a') dx = -1; if (e.key === 'ArrowRight' || e.key === 'd') dx = 1; if (dx || dy) movePlayer(dx, dy); } function startGame() { spawnLevel(); draw(); updateHUD(); logMsg("Welcome to Skibidi Dungeons!"); } document.addEventListener('keydown', handleKey); startGame();
